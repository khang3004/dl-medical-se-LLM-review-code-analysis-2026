\begin{frame}{I. Important Term}
    \begin{columns}[T]
        \begin{column}{0.50\textwidth}
            \textbf{Leukemia} is a group of cancers originating in the bone marrow that causes the overproduction of abnormal, immature white blood cells (leukocytosis). These cells crowd out healthy blood cells, impairing immunity and oxygen transport.
            
            \vspace{0.4cm}
            
            \textbf{Four main types:}
            \begin{itemize}
                \item Acute Myeloid Leukemia (AML)
                \item Chronic Myeloid Leukemia (CML)
                \item Acute Lymphocytic Leukemia (ALL)
                \item Chronic Lymphocytic Leukemia (CLL)
            \end{itemize}
        \end{column}
        \begin{column}{0.47\textwidth}

\end{frame}

\begin{frame}{I. Important Term}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \textbf{GCLI: Global and Local Contexts Interactions}
            
            \vspace{0.4cm}
            
            \onslide<2->{
            \textbf{Local context} describes:
            \begin{itemize}
                \item Shape of individual cells, nuclei (large or small)
                \item Nucleus/cytoplasm ratio
                \item Chromatin texture
            \end{itemize}
            }
            
            \vspace{0.3cm}
            
            \onslide<1->{
            \textbf{Global context} indicates:
            \begin{itemize}
                \item Cell density
                \item Abnormal distribution across the image
                \item Surrounding environment
                \item Multiple blast cells simultaneously
            \end{itemize}
            }
        \end{column}
