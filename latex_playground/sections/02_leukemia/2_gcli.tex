\begin{frame}{The GCLI Solution: Global Context \& Local Interaction}
    \centering
    \vspace{-0.2cm}

    % Define colors locally for safety
    \definecolor{TechBlue}{RGB}{0, 102, 204}
    \definecolor{TechRed}{RGB}{204, 0, 0}
    \definecolor{TechGreen}{RGB}{0, 153, 76}

    \begin{columns}[T]
        % --- Left: The Problem (C2PSA Limitations) ---
        \begin{column}{0.48\textwidth}
            \begin{alertblock}{The Bottleneck: C2PSA (Self-Attention)}
                \centering
                \footnotesize
                \begin{itemize}
                    \item \textbf{Heavy Computation:} Quadratic complexity ($O(N^2)$).
                    \item \textbf{Redundancy:} Processes \textit{all} channels, including noise.
                    \item \textbf{Lack of Interaction:} Local & Global features don't "talk" enough.
                \end{itemize}
            \end{alertblock}
        \end{column}

        % --- Right: The Solution (GCLI) ---
        \begin{column}{0.48\textwidth}
            \begin{block}{The Upgrade: GCLI Module}
                \centering
                \footnotesize
                \begin{itemize}
                    \item \textbf{Partial Utilization:} Learns only $C/2$ channels (Lightweight).
                    \item \textbf{No Dim Reduction:} Uses \textbf{1D Conv} instead of MLP.
                    \item \textbf{Dual Scope:} Captures Long-range dependencies + Local details.
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}

    \vspace{0.5cm}
