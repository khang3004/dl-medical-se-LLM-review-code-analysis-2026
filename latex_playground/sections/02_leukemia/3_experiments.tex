\begin{frame}{IV. Remarks and Conclusion}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Remarks}
            
            \vspace{0.3cm}
            
            \begin{itemize}
                \item Experiments conducted only on \textbf{YOLOv11} (single-stage detector)
                
                \vspace{0.2cm}
                
                \item Future work:
                \begin{itemize}
                    \item Other YOLO variants
                    \item More robust attention mechanisms
                \end{itemize}
                
                \vspace{0.2cm}
                
                \item \textbf{Limitation:} Low sensitivity in identifying latent malignant cells
            \end{itemize}
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \textbf{Conclusion}
            
            \vspace{0.3cm}
            
            \begin{itemize}
                \item \textbf{GCLI} achieves best mAP50-95 (25.1\%)
                
                \vspace{0.2cm}
                
                \item \textbf{CBAM} achieves highest sensitivity (55.4\%)
                
                \vspace{0.2cm}
                
                \item Attention mechanisms improve detection with minimal computational cost
                
                \vspace{0.2cm}
                
                \item Promising direction for automated leukemia diagnosis
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{IV. Remarks and Conclusion}
    \textbf{Conclusions}
    
    \vspace{0.5cm}
    
    \begin{enumerate}
        \item A new \textbf{YOLOv11 architecture with GCLI} replacing C2PSA improves leukemia detection over the baseline and other attention-based variants.
        
        \vspace{0.4cm}
        
        \item The method achieves \textbf{25.1\% mAP50--95} with over \textbf{11\% fewer parameters}, showing a good accuracy--efficiency trade-off.
        
        \vspace{0.4cm}
        
        \item The results demonstrate the \textbf{potential of GCLI-based attention} for improving CAD systems in biomedical imaging.
    \end{enumerate}
\end{frame}
